Formação e Cálculos de Matrizes

Programa em C: Matrizes

#include <stdio.h>

int main() {
  int escolha;
  int i, j;

  // Título e créditos
  printf("****************************************************\n");
  printf("**  Programa para formação e cálculos de matrizes **\n");
  printf("**        Direitos Autorais: Matheus Barbosa da Silva **\n");
  printf("****************************************************\n\n");

  // Menu de seleção de matriz
  printf("Escolha o tamanho da matriz:\n");
  printf("1. Matriz 1x1\n");
  printf("2. Matriz 2x2\n");
  printf("3. Matriz 3x3\n");
  printf("4. Matriz 4x4\n");
  printf("Digite sua escolha (1-4): ");
  scanf("%d", &escolha);

  switch (escolha) {
  case 1: {
    int matriz1[1][1];
    printf("\nDigite o valor para a matriz 1x1: ");
    scanf("%d", &matriz1[0][0]);

    printf("\nFormação da matriz 1x1:\n");
    printf("[%d]\n", matriz1[0][0]);

    printf("\nResultado final: %d\n", matriz1[0][0]);
    break;
  }
  case 2: {
    int matriz2[2][2];
    printf("\nDigite os valores para a matriz 2x2:\n");
    for (i = 0; i < 2; i++) {
      for (j = 0; j < 2; j++) {
        printf("Elemento [%d][%d]: ", i + 1, j + 1);
        scanf("%d", &matriz2[i][j]);
      }
    }

    printf("\nFormação da matriz 2x2:\n");
    for (i = 0; i < 2; i++) {
      for (j = 0; j < 2; j++) {
        printf("%d ", matriz2[i][j]);
      }
      printf("\n");
    }

    int resultado2 = 0;
    for (i = 0; i < 2; i++) {
      for (j = 0; j < 2; j++) {
        resultado2 += matriz2[i][j];
      }
    }
    printf("\nResultado final: %d\n", resultado2);
    break;
  }
  case 3: {
    int matriz3[3][3];
    printf("\nDigite os valores para a matriz 3x3:\n");
    for (i = 0; i < 3; i++) {
      for (j = 0; j < 3; j++) {
        printf("Elemento [%d][%d]: ", i + 1, j + 1);
        scanf("%d", &matriz3[i][j]);
      }
    }

    printf("\nFormação da matriz 3x3:\n");
    for (i = 0; i < 3; i++) {
      for (j = 0; j < 3; j++) {
        printf("%d ", matriz3[i][j]);
      }
      printf("\n");
    }

    int resultado3 = 0;
    for (i = 0; i < 3; i++) {
      for (j = 0; j < 3; j++) {
        resultado3 += matriz3[i][j];
      }
    }
    printf("\nResultado final: %d\n", resultado3);
    break;
  }
  case 4: {
    int matriz4[4][4];
    printf("\nDigite os valores para a matriz 4x4:\n");
    for (i = 0; i < 4; i++) {
      for (j = 0; j < 4; j++) {
        printf("Elemento [%d][%d]: ", i + 1, j + 1);
        scanf("%d", &matriz4[i][j]);
      }
    }

    printf("\nFormação da matriz 4x4:\n");
    for (i = 0; i < 4; i++) {
      for (j = 0; j < 4; j++) {
        printf("%d ", matriz4[i][j]);
      }
      printf("\n");
    }

    int resultado4 = 0;
    for (i = 0; i < 4; i++) {
      for (j = 0; j < 4; j++) {
        resultado4 += matriz4[i][j];
      }
    }
    printf("\nResultado final: %d\n", resultado4);
    break;
  }
  default:
    printf("Escolha inválida! Tente novamente.\n");
    break;
  }

  return 0;
}
