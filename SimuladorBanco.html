Programa em C: Simulador de Banco

Veja o código abaixo e insira os dados necessários para testar o simulador de banco.

1. #include <stdio.h>
2.
3. #define MAX_TENTATIVAS 3
4.
5. typedef struct {
6.    char nome[100];
7.    int numeroConta;
8.    float saldo;
9. } ContaBancaria;
10.
11. void exibirMenu() {
12.     printf("\n-------- SIMULADOR DE BANCO --------\n");
13.     printf("1. Consultar saldo\n");
14.     printf("2. Fazer depósito\n");
15.     printf("3. Fazer retirada\n");

20.
void consultarSaldo(ContaBancaria *conta) {
    printf("\n** Consultando Saldo **\n");
    printf("Saldo atual de %s (Conta #%d): R$ %.2f\n", conta->nome, conta->numeroConta, conta->saldo);
}

void fazerDeposito(ContaBancaria *conta) {
    float valor;
    printf("\n** Fazer Depósito **\n");
    printf("Digite o valor a ser depositado: R$ ");
    scanf("%f", &valor);
    
    if (valor > 0) {
        conta->saldo += valor;
        printf("Depósito de R$ %.2f realizado com sucesso!\n", valor);
    } else {
        printf("Valor inválido para depósito!\n");
    }
}

void fazerRetirada(ContaBancaria *conta) {
    float valor;
    printf("\n** Fazer Retirada **\n");
    printf("Digite o valor a ser retirado: R$ ");
    scanf("%f", &valor);
    
    if (valor > 0 && valor <= conta->saldo) {
        conta->saldo -= valor;
        printf("Retirada de R$ %.2f realizada com sucesso!\n", valor);
    } else {
        printf("Valor inválido ou saldo insuficiente para a retirada!\n");
    }
}

int main() {
    ContaBancaria conta;
    
    printf("Informe o nome do titular da conta: ");
    fgets(conta.nome, sizeof(conta.nome), stdin);
    conta.nome[strcspn(conta.nome, "\n")] = '\0';

    printf("Informe o número da conta: ");
    scanf("%d", &conta.numeroConta);
    
    conta.saldo = 0.0f;
    
    int tentativa = 0;
    int escolha;

    do {
        exibirMenu();
        scanf("%d", &escolha);

        switch (escolha) {
            case 1:
                consultarSaldo(&conta);
                break;
            case 2:
                fazerDeposito(&conta);
                break;
            case 3:
                fazerRetirada(&conta);
                break;
            case 4:
                printf("Saindo... Obrigado por usar o Simulador de Banco!\n");
                break;
            default:
                printf("Opção inválida! Tente novamente.\n");
                tentativa++;
                break;
        }

        if (tentativa >= MAX_TENTATIVAS) {
            printf("Número máximo de tentativas excedido. Saindo...\n");
            break;
        }

    } while (escolha != 4);
    
    return 0;
}
